let guessedNumEl = document.getElementById("guessedNum");
let resultEl = document.getElementById("result");
let checkBtnEL = document.getElementById("checkBtn");
let resetBtnEl = document.getElementById("resetBtn");

let randomNum = Math.ceil(Math.random()*100);

let counter = 0;

console.log( randomNum );

function OnCheckResult(){
    counter += 1;
    let val1 = parseInt( guessedNumEl.value );

    if( counter > 5 ){

        resultEl.textContent = "Game Over!!!!";
        checkBtnEL.style.display = "none";
        resetBtnEl.style.display = "inline-block";
    }
    else{

        if( val1 > randomNum ){

        resultEl.textContent = "Its too High!!!!";
        resultEl.style.color = "red";
    }
    else if ( val1 < randomNum ){

        resultEl.textContent = "Its too Low!!!!";
        resultEl.style.color = "red";
    }
    else{

        resultEl.textContent = `Hurray you win the game you guessed it in ${counter} attempts!!!`;
        resultEl.style.color = "green";
        checkBtnEL.style.display = "none";
        resetBtnEl.style.display = "inline-block";
    }
    }

    guessedNumEl.value = "";
}


function onReset(){

    randomNum = Math.ceil(Math.random()*100);
    counter = 0;
    resultEl.textContent = "";
    resetBtnEl.style.display = "none";
    checkBtnEL.style.display = "inline-block";

    console.log( randomNum );
}
